238  mesg = 'snmp-server host {0}'.format(addr)

242  mesg += ' user {0}'.format(param['user'])
243 
244  # traps/informs
245  mode = ent['tag'][0]
246  if mode == 'informNotify':
247  mesg += ' informs'
248  elif 'user' in param:
249  mesg += ' traps'
250 
251  # noauth/auth/priv
252  if 'user' in param:
253  auth = param['security-level']
254  if auth == 'no-auth-no-priv':
255  mesg += ' noauth'
256  elif auth == 'auth-no-priv':
257  mesg += ' auth'
258  elif auth == 'auth-priv':
259  mesg += ' priv'
260 
261  # retries (informs only)
262  if (mode == 'informNotify') and (ent['retries'] != 3):
263  mesg += ' retries {0}'.format(ent['retries'])
264 
265  # timeout (informs only)
266  if (mode == 'informNotify') and (ent['timeout'] != 1500):
267  mesg += ' timeout {0}'.format(ent['timeout'] / 100)
268 
269  # port
270  if port != 162:
271  mesg += ' port {0}'.format(port)
272 
273  # interface
274  if len(ent['tag']) > 1:
275  mesg += ' interface {0}'.format(ent['tag'][1])
276 
277  break
278 
279  if mesg is None:
280  continue;
281 
282  resp.append(mesg)
