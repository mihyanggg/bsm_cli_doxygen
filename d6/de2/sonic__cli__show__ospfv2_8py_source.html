327 def ospfv2_filter_neighbors_by_neighbor_id(response, intfname):
328  count = 0
329  if 'openconfig-network-instance:areas' in response and 'area' in response['openconfig-network-instance:areas']:
330  for i in range(len(response['openconfig-network-instance:areas']['area'])):
331  areainfo = response['openconfig-network-instance:areas']['area'][i]
332  if 'interfaces' in areainfo and 'interface' in areainfo['interfaces']:
333  for j in range(len(areainfo['interfaces']['interface'])):
334  interfaceinfo = areainfo['interfaces']['interface'][j]
335  if 'openconfig-ospfv2-ext:neighbors-list' in interfaceinfo and 'neighbor' in interfaceinfo['openconfig-ospfv2-ext:neighbors-list']:
336  temp_nbr_list = []
337  while interfaceinfo['openconfig-ospfv2-ext:neighbors-list']['neighbor']:
338  nbr = interfaceinfo['openconfig-ospfv2-ext:neighbors-list']['neighbor'].pop()
339  if 'neighbor-id' in nbr and nbr['neighbor-id'] == intfname:
340  temp_nbr_list.append(nbr)
341  while temp_nbr_list:
342  interfaceinfo['openconfig-ospfv2-ext:neighbors-list']['neighbor'].append(temp_nbr_list.pop())
343  count = count + 1
344  return count
