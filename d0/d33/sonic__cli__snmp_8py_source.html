254 def getEngineID():
255  """ Construct SNMP engineID from the configured value or from scratch """
256  keypath = cc.Path('/restconf/data/ietf-snmp:snmp/engine/engine-id')
257  response=aa.get(keypath)
258 
259  # First, try to get engineID via rest
260  engineID = ''
261  if response.ok() and response.content is not None and len(response.content) != 0:
262  content = response.content
263  if content.has_key('ietf-snmp:engine-id'):
264  engineID = content['ietf-snmp:engine-id']
265  elif content.has_key('engine-id'):
266  engineID = content['engine-id']
267  engineID = engineID.encode('ascii')
268  engineID = engineID.translate(None, ':')
269 
270  # ensure engineID is properly formatted before use. See RFC 3411
271  try:
272  # must be hex (base 16)
273  value = int(engineID, 16)
274  # length of 5 - 32 octets
275  if len(engineID) < 10:
276  engineID = ''
277  if len(engineID) > 64:
278  engineID = ''
279  except:
280  # Whoops, not hex
281  engineID = ''
282 
283  # if the engineID is not configured, construct as per SnmpEngineID
284  # TEXTUAL-CONVENTION in RFC 3411 using the system MAC address.
285  if len(engineID) == 0:
286  keypath = cc.Path('/restconf/data/sonic-device-metadata:sonic-device-metadata/DEVICE_METADATA/DEVICE_METADATA_LIST={name}/mac', name="localhost")
287  response = aa.get(keypath, None, False)
288  if response.ok():
289  sysmac = response.content['sonic-device-metadata:mac'].encode('ascii')
290  if sysmac == None:
291  # All else fails, something must be used. Fabricated MAC Address
292  sysmac = '00:00:00:12:34:56'
293  sysmac = sysmac.translate(None, ':')
294  # engineID is:
295  # 3) The length of the octet string varies.
296  # bit 0 == '1'
297  # The snmpEngineID has a length of 12 octets
298  # The first four octets are set to the binary equivalent of the agent's
299  # SNMP management private enterprise number as assigned by IANA.
300  # Microsoft = 311 = 0000 0137
301  # The fifth octet indicates how the rest (6th andfollowing octets) are formatted.
302  # 3 - MAC address (6 octets)
303  # + System MAC address
304  engineID = "8000013703"+sysmac
305 
306  return engineID
