73  intfSwitch = '-i'

75  stdNaming = is_intf_naming_mode_std()

77  # Check for interfaces, order is important.
78  # Regex for subinterfaces should come first.
79 
80  #Check for subinterface
81  result = re.search('('+intfSwitch+'\s*)(Ethernet|PortChannel)(\s*)(\d+\.\d+)(\s+|$)', args, re.IGNORECASE)
82  if result:
83  if stdNaming: #For std naming we shouldn't come here
84  return None, INVALID_IFNAME
85  if result.group(2).startswith("Eth"):
86  ifName = "Eth" + result.group(4)
87  elif result.group(2).startswith("Port"):
88  ifName = "Po" + result.group(4)
89  return ifName, NATIVE_SUBIFNAME
90 
91  #Check for subinterface with alias
92  result = re.search('('+intfSwitch+'\s*)(Eth)(\s*)(\d+/\d+/*\d*\.\d+)(\s+|$)', args, re.IGNORECASE)
93  if result:
94  if not stdNaming:
95  return None, INVALID_IFNAME
96  if result.group(2).startswith("Eth"):
97  ifName = "Eth" + result.group(4)
98  ifName = get_alias(ifName)
99  return ifName, STD_SUBIFNAME
100 
101  #Check for interface
102  result = re.search('('+intfSwitch+'\s*)(Ethernet)(\s*)(\d+)(\s+|$)', args, re.IGNORECASE)
103  if result:
104  if stdNaming: #For std naming we shouldn't come here
105  return None, INVALID_IFNAME
106  ifName = result.group(2) + result.group(4)
107  return ifName, NATIVE_IFNAME
108 
109  #Check for interface with alias
110  result = re.search('('+intfSwitch+'\s*)(Eth)(\s*)(\d+/\d+/*\d*)(\s+|$)', args, re.IGNORECASE)
111  if result:
112  if not stdNaming:
113  return None, INVALID_IFNAME
114  ifName = result.group(2) + result.group(4)
115  ifName = get_alias(ifName)
116  return ifName, STD_IFNAME
117 
118  #Check for mgmt interface
119  result = re.search('('+intfSwitch+'\s*)(Management)(\s*)(\d+)(\s+|$)', args, re.IGNORECASE)
120  if result:
121  ifName = "eth" + result.group(4)
